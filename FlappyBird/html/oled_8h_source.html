107 /* Unlocks the busy lock (independent of its original condition) */
108 inline ALWAYSINLINE void OLED_unlock(OLED *oled)
109 {
110  oled->busy_lock = 1;
111  /* This is translated to: */
112  /* ldi r0, lo8(1) */
113  /* movw r30, r24 */
114  /* std Z+6, r0 */

118 /* Tries to atomically acquire lock. On success captures lock and returns
119  * true. On failure does nothing and returns false.
120  * Usage is like:
121  * if (OLED_trylock(&myoled)) {
122  * ...do something...
123  * OLED_unlock(&myoled)
124  * } else {
125  * ...OLED is busy. Do other stuff...
126  * }
127  */
128 #ifdef __AVR_XMEGA__
129 inline ALWAYSINLINE bool OLED_trylock(OLED *oled)
130 {
131  /* Relies on assembly directives. For more, see: */
132  /* https://www.nongnu.org/avr-libc/user-manual/inline_asm.html */
133  /* And LAC in AVR Instruction Set Manual */
134  /* %0 is Z register (ptr is read-only, destination is read-written) */
135  /* %1 is r0...r31 (read-write) */
136  /* Works like: (but is atomic, single instruction) */
137  /* lock_t old = busy_lock; */
138  /* busy_lock = busy_lock & ~val; */
139  /* val = old; */
140 
141  bool val = true;
142  asm volatile("lac %0, %A1" :
143  "z" (&oled->busy_lock) :
144  "+r" (val));
145  return !val;
146 }
147 #else
148 inline ALWAYSINLINE bool OLED_trylock(OLED *oled)
149 {
150  /* We do not have atomic instructions on non-XMEGA cores, so try to */
151  /* stick with what's available */
152 
153  bool val;
154  asm volatile(
155  /* Read SREG containing Interrupts Enabled flag to tmp reg */
156  "in __tmp_reg__, __SREG__\n\t"
157  /* Disable interrupts. Interrupt can occur between in and */
158  /* cli. It couldn't be fixed because of cpu architecture */
159  "cli\n\t"
160  "ld %0, %a1\n\t"
161  /* Write 0 to lock */
162  "st %a1, __zero_reg__\n\t"
163  /* Restore original SREG with Interrupts Enabled flag */
164  "out __SREG__, __tmp_reg__" :
165  /* Attributes below */
166  "=&r" (val) :
167  "z" (&oled->busy_lock)
168  );
169  return !val;
170 }
171 #endif

174 /* Cycles till the lock is unlocked, acquires it and only then exits
175  * (!) Warning: may cause deadlock (infinite wait for resource to free)
176  */
177 inline ALWAYSINLINE bool OLED_spinlock(OLED *oled)
178 {
179  while (!OLED_trylock(oled));
180  return true;
181 }

184 /* Used as a header of block to execute its contents with OLED_spinlock
185  * (!) Warning: may cause deadlock (infinite wait for resource to free)
186  * Usage example:
187  * OLED_WITH_SPINLOCK(&oled) {
188  * ...do something...
189  * }
190  * In a for(a; b; c) { ...body... } first a is executed. Then b is tested. Then
191  * body is run. Then c is executed. After this, b is tested (and this time)
192  * it will evaluate as false.
193  * For more see: https://www.tutorialspoint.com/cprogramming/c_for_loop.htm
194  */
195 #define OLED_WITH_SPINLOCK(oled) for(bool __tmp = OLED_spinlock((oled)); __tmp; OLED_unlock((oled)), __tmp = false)

199  * Usage example:
200  * bool isrun = false;
201  * OLED_WITH_TRYLOCK(&oled, isrun) {
202  * ...do something...
203  * }
204  * // now isrun is true if trylock was successful
205  * Or:
206  * OLED_WITH_TRYLOCK(&oled) {

249 /* Inline put pixel, without checks. See the full method below */
250 /* Used to allow GCC to optimize other draw routines which use put_pixel */
251 inline ALWAYSINLINE void OLED_put_pixel_(OLED *oled, uint8_t x, uint8_t y, bool pixel_state)
252 {
253  /* Find byte index in flat array */
254  uint16_t byte_num = (y / 8) * (uint16_t)oled->width + x;
255  uint8_t bit_y = y % 8;
256  if (pixel_state)
257  oled->frame_buffer[byte_num] |= (1 << bit_y);
258  else
259  oled->frame_buffer[byte_num] &= ~(1 << bit_y);
260 }
